import React, { useState, useRef } from 'react';
import { View, Modal, Text, Button } from 'react-native';
import Video from 'react-native-video';

const App = () => {
  const videoRef = useRef(null);
  const [isModalVisible, setModalVisible] = useState(false);

  // Event handler for when the video is progressing
  const onProgress = (data) => {
    const currentTime = data.currentTime;

    // Check if it's the specific time you want to pause and show the modal
    if (currentTime >= YOUR_SPECIFIC_TIME_IN_SECONDS) {
      videoRef.current.pause();
      setModalVisible(true);
    }
  };

  // Close the modal and resume video playback
  const closeModal = () => {
    setModalVisible(false);
    videoRef.current.seek(YOUR_SPECIFIC_TIME_IN_SECONDS); // Optional: Seek to a specific time if needed
    videoRef.current.play(); // Optional: Resume playback
  };

  return (
    <View style={{ flex: 1 }}>
      <Video
        ref={videoRef}
        source={{ uri: 'your_video_source_here.mp4' }}
        onProgress={onProgress}
        // Add other video props as needed
      />

      {/* Modal to be displayed at a specific time */}
      <Modal visible={isModalVisible} transparent animationType="slide">
        <View style={{ flex: 1, justifyContent: 'center', alignItems: 'center' }}>
          <View style={{ backgroundColor: 'white', padding: 20, borderRadius: 10 }}>
            <Text>This is your modal content</Text>
            <Button title="Close Modal" onPress={closeModal} />
          </View>
        </View>
      </Modal>
    </View>
  );
};

export default App;
